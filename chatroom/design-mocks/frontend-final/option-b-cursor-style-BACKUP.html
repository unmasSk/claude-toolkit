export function checkRateLimit(
  ws: ServerWebSocket,
  config: RateLimitConfig
): boolean {
  const now = Date.now();
  const window = ws.data.requests
    .filter(t => t > now - config.windowMs);
  if (window.length >= config.maxRequests) {
    ws.send(JSON.stringify({
      error: 'Rate limit exceeded',
      code: 'RATE_LIMITED',
      retryAfter: Math.ceil(
        (window[0] + config.windowMs - now) / 1000
      )
    }));
    return false;
  }
  ws.data.requests = [...window, now];
  return true;
}

Diagnosis: Init-Order Race in WebSocket Upgrade
Bilbo is right but the root cause is deeper. The token bucket isn't just getting defaults — it's getting someone else's defaults. Look at the constructor:
The TokenBucket constructor reads from a module-level _defaultConfig that gets mutated by the last room that called configureRoom(). This is shared mutable state across all connections.
Fix order:
  1. Make _defaultConfig a frozen copy per room, not a shared reference
  2. Call session.init(roomConfig) synchronously in the upgrade handler
  3. Add regression test: two rooms with different rate limits connecting simultaneously

Security Warning: The shared mutable _defaultConfig is not just a bug — it's an attack surface. A malicious client connecting to a permissive room can pollute the rate-limit config for strict rooms, effectively bypassing throttling for all subsequent connections.
exploit
// Attacker connects to "open-lobby" (100 msg/10s limit)
// _defaultConfig is now { maxMessages: 100, window: 10000 }
// Next connection to "secure-room" (5 msg/10s limit)
// inherits open-lobby's config → bypass achieved
ws.connect("wss://host/open-lobby");  // pollute config
ws.close();
ws.connect("wss://host/secure-room"); // 100 msg/10s now
