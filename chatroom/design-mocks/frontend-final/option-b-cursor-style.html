export function checkRateLimit(
  ws: ServerWebSocket,
  config: RateLimitConfig
): boolean {
  const now = Date.now();
  const window = ws.data.requests
    .filter(t => t > now - config.windowMs);
  if (window.length >= config.maxRequests) {
    ws.send(JSON.stringify({
      error: 'Rate limit exceeded',
      code: 'RATE_LIMITED',
      retryAfter: Math.ceil(
        (window[0] + config.windowMs - now) / 1000
      )
    }));
    return false;
  }
  ws.data.requests = [...window, now];
  return true;
}
